why prefect pipeline takes time? also i need to submit my checkpoint presentation in two days is is it doable

Thought for a couple of seconds
Yes — checkpoint in 2 days is doable.

And Prefect takes time mostly because it is not just writing Python functions. You have to make your pipeline behave like a real workflow system.

Why Prefect usually takes time
1. You have to split code into tasks
A normal script is:

download_data()
clean_data()
save_data()
train_model()
In Prefect, you usually turn these into separate @tasks and connect them inside a @flow. That restructuring itself takes time.

2. Debugging is different
Code that works in a notebook may fail in a flow because of:

wrong paths

missing imports

return values not passed correctly

task dependencies

config/environment issues

3. Retries and error handling
Your guidelines specifically ask for:

retries

failure handling

scheduling

observable runs/logs 


That means you cannot stop at “it runs once.”
You have to add a bit of workflow logic.

4. Scheduling and orchestration setup
Even a simple scheduled run needs extra setup compared to a plain script.
That is why Prefect often becomes a time sink.

5. New tool overhead
If you already know pandas, APIs, and notebooks, those parts go faster.
Prefect is slower because it is usually newer and more “system-like.”

import os
import sqlite3
from datetime import datetime

import pandas as pd
import requests
from dotenv import load_dotenv


load_dotenv()

API_KEY = os.getenv("NEWS_API_KEY")
URL = "https://newsapi.org/v2/top-headlines"


def fetch_top_headlines(country="us", category="technology", page_size=20):
    if not API_KEY:
        raise ValueError("NEWS_API_KEY not found in .env file")

    params = {
        "country": country,
        "category": category,
        "pageSize": page_size,
        "apiKey": API_KEY
    }

    response = requests.get(URL, params=params, timeout=30)
    response.raise_for_status()
    data = response.json()

    if data.get("status") != "ok":
        raise ValueError(f"API error: {data}")

    articles = data.get("articles", [])
    fetched_at = datetime.utcnow().isoformat()

    rows = []
    for article in articles:
        rows.append({
            "source_name": article.get("source", {}).get("name"),
            "author": article.get("author"),
            "title": article.get("title"),
            "description": article.get("description"),
            "url": article.get("url"),
            "published_at": article.get("publishedAt"),
            "content": article.get("content"),
            "category": category,
            "fetched_at": fetched_at
        })

    return pd.DataFrame(rows)


def save_to_sqlite(df, db_path="data/news.db"):
    os.makedirs("data", exist_ok=True)
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()

    cursor.execute("""
    CREATE TABLE IF NOT EXISTS articles (
        article_id INTEGER PRIMARY KEY AUTOINCREMENT,
        source_name TEXT,
        author TEXT,
        title TEXT,
        description TEXT,
        url TEXT UNIQUE,
        published_at TEXT,
        content TEXT,
        category TEXT,
        fetched_at TEXT
    )
    """)

    for _, row in df.iterrows():
        cursor.execute("""
        INSERT OR IGNORE INTO articles
        (source_name, author, title, description, url, published_at, content, category, fetched_at)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            row["source_name"],
            row["author"],
            row["title"],
            row["description"],
            row["url"],
            row["published_at"],
            row["content"],
            row["category"],
            row["fetched_at"]
        ))

    conn.commit()
    conn.close()


if __name__ == "__main__":
    df = fetch_top_headlines(country="us", category="technology", page_size=20)
    print(df.head())
    save_to_sqlite(df)
    print(f"Saved {len(df)} articles to SQLite")

# API endpoint
URL = "https://newsapi.org/v2/top-headlines"


def fetch_top_headlines(country="us", category="technology", page_size=20):
    if not API_KEY:
        raise ValueError("NEWS_API_KEY not found in .env file")

    params = {
        "country": country,
        "category": category,
        "pageSize": page_size,
        "apiKey": API_KEY
    }

    response = requests.get(URL, params=params, timeout=30)
    response.raise_for_status()

    data = response.json()

    if data.get("status") != "ok":
        raise ValueError(f"API error: {data}")

    articles = data.get("articles", [])
    fetched_at = datetime.utcnow().isoformat()

    rows = []
    for article in articles:
        rows.append({
            "source_name": article.get("source", {}).get("name"),
            "author": article.get("author"),
            "title": article.get("title"),
            "description": article.get("description"),
            "url": article.get("url"),
            "published_at": article.get("publishedAt"),
            "content": article.get("content"),
            "category": category,
            "fetched_at": fetched_at
        })

    return pd.DataFrame(rows)


def save_to_sqlite(df, db_path="data/news.db"):
    os.makedirs("data", exist_ok=True)

    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()

    # Create table
    cursor.execute("""
    CREATE TABLE IF NOT EXISTS articles (
        article_id INTEGER PRIMARY KEY AUTOINCREMENT,
        source_name TEXT,
        author TEXT,
        title TEXT,
        description TEXT,
        url TEXT UNIQUE,
        published_at TEXT,
        content TEXT,
        category TEXT,
        fetched_at TEXT
    )
    """)

    # Insert data (ignore duplicates using UNIQUE url)
    for _, row in df.iterrows():
        cursor.execute("""
        INSERT OR IGNORE INTO articles
        (source_name, author, title, description, url, published_at, content, category, fetched_at)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            row["source_name"],
            row["author"],
            row["title"],
            row["description"],
            row["url"],
            row["published_at"],
            row["content"],
            row["category"],
            row["fetched_at"]
        ))

    conn.commit()
    conn.close()

def fetch_news(query="technology OR AI OR business OR gaming", language="en", page_size=20):
    if not API_KEY:
        raise ValueError("NEWS_API_KEY not found in .env file")

    params = {
        "q": query,
        "language": language,
        "pageSize": page_size,
        "sortBy": "publishedAt",
        "apiKey": API_KEY
    }

    response = requests.get(URL, params=params, timeout=30)
    response.raise_for_status()

    data = response.json()

    if data.get("status") != "ok":
        raise ValueError(f"API error: {data}")

    articles = data.get("articles", [])
    fetched_at = datetime.now(UTC).isoformat()

    rows = []
    for article in articles:
        title = article.get("title")

        rows.append({
            "source_name": article.get("source", {}).get("name"),
            "author": article.get("author"),
            "title": title,
            "description": article.get("description"),
            "url": article.get("url"),
            "published_at": article.get("publishedAt"),
            "content": article.get("content"),
            "category": assign_category(title),
            "fetched_at": fetched_at
        })

    return pd.DataFrame(rows)

import sqlite3
import pandas as pd


DB_PATH = "data/news.db"


def load_data():
    conn = sqlite3.connect(DB_PATH)
    df = pd.read_sql("SELECT * FROM articles", conn)
    conn.close()
    return df


def validate_data(df):
    print("\n===== DATA VALIDATION REPORT =====\n")

    # 1. Shape
    print(f"Total rows: {len(df)}")
    print(f"Total columns: {len(df.columns)}\n")

    # 2. Missing values
    print("Missing values per column:")
    missing = df.isnull().sum()
    print(missing, "\n")

    # 3. Critical column checks
    print("Checking critical columns...")

    null_titles = df["title"].isnull().sum()
    null_urls = df["url"].isnull().sum()

    print(f"Missing titles: {null_titles}")
    print(f"Missing URLs: {null_urls}\n")

    # 4. Duplicate check
    duplicate_urls = df["url"].duplicated().sum()
    print(f"Duplicate URLs: {duplicate_urls}\n")

    # 5. Category distribution
    print("Category distribution:")
    print(df["category"].value_counts(), "\n")

    # 6. Basic assertions (acts like validation rules)
    print("Running validation rules...\n")

    if null_urls > 0:
        print("❌ ERROR: Null URLs found")
    else:
        print("✅ URLs are valid")

    if duplicate_urls > 0:
        print("❌ ERROR: Duplicate URLs found")
    else:
        print("✅ No duplicate URLs")

    if null_titles > 5:
        print("⚠️ WARNING: Too many missing titles")
    else:
        print("✅ Titles look fine")

    print("\n===== VALIDATION COMPLETE =====\n")

import os
import sqlite3
from datetime import datetime, UTC

import pandas as pd
import requests
from dotenv import load_dotenv

from logs.logger import setup_logger


logger = setup_logger()


def assign_category(title):
    title = str(title).lower()

    if any(k in title for k in ["ai", "technology", "software", "google", "microsoft", "iphone", "android", "windows"]):
        return "technology"
    elif any(k in title for k in ["stock", "market", "business", "finance", "economy", "trade", "company"]):
        return "business"
    elif any(k in title for k in ["game", "xbox", "nintendo", "playstation", "gaming", "switch"]):
        return "gaming"
    else:
        return "general"


load_dotenv()
API_KEY = os.getenv("NEWS_API_KEY")

URL = "https://newsapi.org/v2/everything"


def fetch_news(query="technology OR AI OR business OR gaming", language="en", page_size=20):
    if not API_KEY:
        raise ValueError("NEWS_API_KEY not found in .env file")

    params = {
        "q": query,
        "language": language,
        "pageSize": page_size,
        "sortBy": "publishedAt",
        "apiKey": API_KEY
    }

    logger.info("Sending request to NewsAPI")
    response = requests.get(URL, params=params, timeout=30)
    response.raise_for_status()

    data = response.json()

    if data.get("status") != "ok":
        raise ValueError(f"API error: {data}")

    articles = data.get("articles", [])
    fetched_at = datetime.now(UTC).isoformat()

    rows = []
    for article in articles:
        title = article.get("title")

        rows.append({
            "source_name": article.get("source", {}).get("name"),
            "author": article.get("author"),
            "title": title,
            "description": article.get("description"),
            "url": article.get("url"),
            "published_at": article.get("publishedAt"),
            "content": article.get("content"),
            "category": assign_category(title),
            "fetched_at": fetched_at
        })

    df = pd.DataFrame(rows)
    logger.info(f"Fetched {len(df)} articles from API")
    return df


def save_to_sqlite(df, db_path="data/news.db"):
    os.makedirs("data", exist_ok=True)

    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()

    cursor.execute("""
    CREATE TABLE IF NOT EXISTS articles (
        article_id INTEGER PRIMARY KEY AUTOINCREMENT,
        source_name TEXT,
        author TEXT,
        title TEXT,
        description TEXT,
        url TEXT UNIQUE,
        published_at TEXT,
        content TEXT,
        category TEXT,
        fetched_at TEXT
    )
    """)

    inserted = 0
    for _, row in df.iterrows():
        cursor.execute("""
        INSERT OR IGNORE INTO articles
        (source_name, author, title, description, url, published_at, content, category, fetched_at)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            row["source_name"],
            row["author"],
            row["title"],
            row["description"],
            row["url"],
            row["published_at"],
            row["content"],
            row["category"],
            row["fetched_at"]
        ))
        inserted += cursor.rowcount

    conn.commit()
    conn.close()

    logger.info(f"Inserted {inserted} new rows into SQLite")


if __name__ == "__main__":
    logger.info("Starting news ingestion")

    try:
        df = fetch_news(
            query="technology OR AI OR business OR gaming",
            language="en",
            page_size=20
        )

        logger.info("Preview of fetched data:")
        logger.info("\n%s", df.head())

        save_to_sqlite(df)

        logger.info("News ingestion completed successfully")

    except Exception as e:
        logger.error(f"Error in ingestion: {e}")
        raise

import sqlite3
import pandas as pd

from logs.logger import setup_logger


logger = setup_logger()
DB_PATH = "data/news.db"


def load_data():
    conn = sqlite3.connect(DB_PATH)
    df = pd.read_sql("SELECT * FROM articles", conn)
    conn.close()
    return df


def validate_data(df):
    logger.info("===== DATA VALIDATION REPORT =====")

    logger.info(f"Total rows: {len(df)}")
    logger.info(f"Total columns: {len(df.columns)}")

    logger.info("Missing values per column:")
    logger.info("\n%s", df.isnull().sum())

    null_titles = df["title"].isnull().sum()
    null_urls = df["url"].isnull().sum()
    duplicate_urls = df["url"].duplicated().sum()

    logger.info("Checking critical columns...")
    logger.info(f"Missing titles: {null_titles}")
    logger.info(f"Missing URLs: {null_urls}")
    logger.info(f"Duplicate URLs: {duplicate_urls}")

    logger.info("Category distribution:")
    logger.info("\n%s", df["category"].value_counts())

    logger.info("Running validation rules...")

    if null_urls > 0:
        logger.error("Null URLs found")
    else:
        logger.info("URLs are valid")

    if duplicate_urls > 0:
        logger.error("Duplicate URLs found")
    else:
        logger.info("No duplicate URLs")

    if null_titles > 5:
        logger.warning("Too many missing titles")
    else:
        logger.info("Titles look fine")

    logger.info("===== VALIDATION COMPLETE =====")


if __name__ == "__main__":
    try:
        df = load_data()
        validate_data(df)
    except Exception as e:
        logger.error(f"Error during validation: {e}")
        raise
